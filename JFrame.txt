    private void jsaveActionPerformed(java.awt.event.ActionEvent evt) {
        if (jenroll.getText().equals("")) {
            JOptionPane.showMessageDialog(null, "Enter enrollnment number");
        } else if (jname.getText().equals("")) {
            JOptionPane.showMessageDialog(null, "Enter name");
        } else if (jemail.getText().equals("")) {
            JOptionPane.showMessageDialog(null, "Enter email");
        } else if (jDateChooser.getDate() == null) {
            JOptionPane.showMessageDialog(null, "Enter date");
        } else if (jmale.isSelected() == false && (jfemale.isSelected() == false)) {
            JOptionPane.showMessageDialog(null, "Enter gender");
        } else if (jdiv.getSelectedItem() == "Select") {
            JOptionPane.showMessageDialog(null, "Enter division");
        } else if (!jdance.isSelected() && !jsing.isSelected() && !jread.isSelected() && !jsport.isSelected()) {
            JOptionPane.showMessageDialog(null, "Enter your interest");
        } else if (jadd.getText().equals("")) {
            JOptionPane.showMessageDialog(null, "Enter address");
        } else {

            try {
                Class.forName("com.mysql.cj.jdbc.Driver");
                Connection cn = DriverManager.getConnection("jdbc:mysql://localhost/3tc", "root", "");
                System.out.println("Connected");
                PreparedStatement ps = cn.prepareStatement("INSERT INTO tblreg1(`Enroll`, `Name`, `Email`, `DOB`, `Gender`, `Division`, `Interest`, `Address`) VALUES(?,?,?,?,?,?,?,?)");
                ps.setInt(1, Integer.parseInt(jenroll.getText()));
                ps.setString(2, jname.getText());
                ps.setString(3, jemail.getText());
                ps.setString(4, ((JTextField) jDateChooser.getDateEditor().getUiComponent()).getText());
                if (jmale.isSelected()) {
                    ps.setString(5, jmale.getText());
                } else {
                    ps.setString(5, jfemale.getText());
                }

                String value = jdiv.getSelectedItem().toString();
                ps.setString(6, value);

                String fav_sub = "";
                if (jdance.isSelected()) {
                    fav_sub += jdance.getText() + " , ";
                }
                if (jsing.isSelected()) {
                    fav_sub += jsing.getText() + " , ";
                }
                if (jread.isSelected()) {
                    fav_sub += jread.getText() + " , ";
                }
                if (jsport.isSelected()) {
                    fav_sub += jsport.getText() + " , ";
                }
                ps.setString(7, fav_sub);

                ps.setString(8, jadd.getText());

                // cleardata();
                int i = ps.executeUpdate();
                JOptionPane.showMessageDialog(jsave, i + " Record Inserted Successfully");
                cleardata();
                dispose();
                setVisible(true);
                ps.close();
                cn.close();
            } catch (Exception e) {
                System.out.println(e);
            }
        }
    }
